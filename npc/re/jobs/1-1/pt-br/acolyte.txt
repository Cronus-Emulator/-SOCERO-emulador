prt_church,184,41,4	script	Cleric#aco	60,{
	if (Upper == 1) {
		if (ADVJOB == Job_High_Priest || ADVJOB == Job_Champion) {
			if (Class == Job_Novice_High) {
				mes "[Padre Mareusis]";
				mes "Ohh, consigo ver que você suportou";
				mes "muita coisa da vida passada. Deve ter aprendido inúmeras coisas antes de chegar em Valhalla.";
				next;
				if (!callfunc("F_CanChangeJob")) {
					mes "[Padre Mareusis]";
					mes "Infelizmente você ainda não está pronto para se tornar Noviço, aprenda as habilidades básicas primeiro e retorne.";
					next;
					mes "[Padre Mareusis]";
					mes "Enquanto isso irei";
					mes "rezar até que você";
					mes "esteja pronto.";
					mes "Que Deus esteja";
					mes "esteja com você.";
					close;
				}
				mes "[Padre Mareusis]";
				mes "Bem, eu te recebo";
				mes "de volta de Valhalla e";
				mes "desejo sorte em";
				mes "sua nova jornada.";
				next;
				skill "NV_TRICKDEAD",0,SKILL_PERM;
				jobchange Job_Acolyte_High;
				skill "AL_HOLYLIGHT",1,SKILL_PERM;
				mes "[Padre Mareusis]";
				mes "Agora, busque aqueles que precisam de sua ajuda. Que Deus esteja com você.";
				close;
			}else{
				mes "[Padre Mareusis]";
				mes "Agora busque aqueles que precisam de sua ajuda. Que Deus esteja com você.";
				close;
			}
		}else{
			mes "[Padre Mareusis]";
			mes "Ohh, consigo ver que você suportou muita coisa da vida passada. Deve ter aprendido inúmeras coisas antes de chegar em Valhalla.";
			next;
			mes "[Padre Mareusis]";
			mes "Mas, por hora, você não está qualificado para ser um noviço. Lembre-se quem você foi na vida passada antes de encontrar seu caminho.";
			close;
		}
	}
	mes "[Padre Mareusis]";
	mes "O que você procura?";
	next;
	switch(select("Virar Noviço.:O que preciso para ser noviço?.:Sair daqui.")) {
	case 1:
		mes "[Padre Mareusis]";
		if(BaseJob != Job_Novice) {
			if(BaseJob == Job_Acolyte)
				mes "Como você está hoje? Vejo que já é um noviço, está brincando né?";
			else
				mes "Me desculpe, mas você não é um noviço";
			close;
		}else{
			mes "Você realmente quer virar um servo de Deus?";
			mes "Veremos se você está preparado para isso... Hmm...";
			next;
			if (!callfunc("F_CanChangeJob")) {
				mes "[Padre Mareusis]";
				mes "Mas o que é isto!? Você não completou nem as habilidades básicas, vá já upar este classe preguiçoso.";
				close;
			}
			mes "[Padre Mareusis]";
			mes "Bom, requisitos cumpridos...";
			mes "Agora irei qualifica-lo para tornar-se um Noviço";
			next;
			mes "[Padre Mareusis]";
			mes "Seja sempre grato a Deus! Nas dificuldades tenha sempre uma fé inabalável.";
			callfunc "Job_Change",Job_Acolyte;
			getitem 1545,1;	// N_Mace
			next;
			mes "[Padre Mareusis]";
			mes "Finalmente... meus sinceros parabéns por preservar-se em sua prova de penitência.";
			close;
		}
	case 2:
		mes "[Padre Mareusis]";
		mes "Você quer ser um noviço?";
		mes "Então deverá seguir as exigências de Deus para isto.";
		next;
		mes "[Padre Mareusis]";
		mes "Primeiro deverá ser Aprendiz, ter pelo menos nível 9 de classe e aprender todas habilidades básicas.";
		mes "Isto é o básico.";
		next;
		mes "[Padre Mareusis]";
		mes "Quando conseguir os requisitos, volte aqui que irei torná-lo noviço. E poderá para sempre espalhar a vontade de Deus.";
		close;
	case 3:
		close;
	}
}
